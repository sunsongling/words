### addPet\.js

```javascript
const app = getApp()
const db = wx.cloud.database()

Page({
  data:{
    isEdit:false,
    petId:'',
    avatar:'',
    petName:'',
    typeList:['猫咪','狗狗'],
    typeIndex:0,
    petAge:''
  },
  onLoad(options){
    if(options.id){
      this.setData({isEdit:true,petId:options.id})
      this.getPetInfo()
    }
  },
  // 获取编辑宠物信息
  async getPetInfo(){
    let res = await db.collection('pet_list').doc(this.data.petId).get()
    let info = res.data
    this.setData({
      avatar:info.pet_avatar,
      petName:info.pet_name,
      typeIndex:this.data.typeList.findIndex(v=>v===info.pet_type),
      petAge:info.pet_age
    })
  },
  // 选择头像
  async chooseAvatar(){
    let res = await wx.chooseImage({count:1,sizeType:['compressed']})
    let path = res.tempFilePaths[0]
    // 上传云存储
    let upload = await wx.cloud.uploadFile({
      cloudPath:`pet_avatar/${Date.now()}.png`,
      filePath:path
    })
    this.setData({avatar:upload.fileID})
  },
  // 输入昵称
  inputName(e){
    this.setData({petName:e.detail.value})
  },
  // 选择类型
  selectType(e){
    this.setData({typeIndex:e.detail.value})
  },
  // 输入年龄
  inputAge(e){
    this.setData({petAge:e.detail.value})
  },
  // 提交保存
  async submit(){
    let {petName,typeIndex,petAge,avatar,isEdit,petId,typeList} = this.data
    if(!petName) return wx.showToast({title:'请输入宠物昵称',icon:'none'})
    if(!petAge) return wx.showToast({title:'请输入宠物年龄',icon:'none'})

    let data = {
      openid:app.globalData.openid,
      pet_name:petName,
      pet_type:typeList[typeIndex],
      pet_age:petAge,
      pet_avatar:avatar||''
    }

    if(isEdit){
      await db.collection('pet_list').doc(petId).update({data})
      wx.showToast({title:'修改成功'})
    }else{
      await db.collection('pet_list').add({data})
      wx.showToast({title:'添加成功'})
    }
    setTimeout(()=>wx.navigateBack(),1000)
  }
})
```
